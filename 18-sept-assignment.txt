int main()
{
    
    int i,j=1,c,temp2;

    gets(st);
    for(i=0;st[i]!='\0';i++)
    {
        c=1;
        for(j=0;st[j]!='\0';j--)
        {
            if(st[i]==st[j])
            {
                count[i]=c++;
            }
        }
    }
    for(i=0;st[i]!='\0';i++)
    {
        for(j=0;j<i;j++)
        {
            if(st[i]==st[j])
            break;
        }
        if(j==i)
        printf("%c-%d\n",st[i],count[i]);
       
    }
      
    return 0;
}
